const Schema = mongoose.Schema({
  name: mongoose.Schema.Types.String,
  email: mongoose.Schema.Types.String,
  password: mongoose.Schema.Types.String,
  score: mongoose.Schema.Types.Number,
  teachers: [mongoose.Schema.Types.ObjectId],
  students: [mongoose.Schema.Types.ObjectId],
  fromteachers: [mongoose.Schema.Types.ObjectId],
  fromstudents: [mongoose.Schema.Types.ObjectId],
  toteachers: [mongoose.Schema.Types.ObjectId],
  tostudents: [mongoose.Schema.Types.ObjectId],

const tasksSchema = mongoose.Schema({
